t_tradegoods = {
	scope = country

  ai_is_valid = {
		always = no
	}

  effect = {
		if = {
			limit = { has_variable = t_mapmode }
			remove_variable = t_mapmode
		}
		trigger_event = { id = map.100 }
  }
}

t_government = {
	scope = country

  ai_is_valid = {
		always = no
	}

  is_shown = {
    has_variable = t_mapmode
		var:t_mapmode = 1 # yes, I hate zeros, bite me
  }

  effect = {
		set_variable = {
			name = t_mapmode
			value = 1
		}
		trigger_event = { id = map.1 }
  }
}

t_armies = {
	scope = country

  ai_is_valid = {
		always = no
	}

  is_shown = {
    has_variable = t_armies
  }

  effect = {
    if = {
      limit = {
        has_variable = t_armies
      }
      remove_variable = t_armies
    }
		else = {
			set_variable = t_armies
		}
  }
}

t_mercs = {
	scope = country

  ai_is_valid = {
		always = no
	}

  is_shown = {
		OR = {
	    has_variable = t_mercs
			has_variable = t_armies
		}
  }

  effect = {
    if = {
      limit = {
        has_variable = t_mercs
      }
      remove_variable = t_mercs
    }
		else = {
			set_variable = t_mercs
		}
  }
}
