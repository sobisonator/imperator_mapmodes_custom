MAPMODE_enable_special_mapmode = {
	trigger_event = map.11 # Store the city ranks
	MAPMODE_enable_mapmode_$mapmode$ = yes
	#trigger_event = map.100 # Clear/reset city ranks
}

MAPMODE_set_city_status = { # Special scripted effect to give the correct equivalent "fake" city ranks
	# Scope: Province
	if = {
		limit = {
			has_variable = province_rank
		}
		if = {
			limit = { var:province_rank = 1 }
			set_city_status = $mapmode_type$_settlement_$number$			
		}
		if = {
			limit = { var:province_rank = 2 }
			set_city_status = $mapmode_type$_city_$number$			
		}
		if = {
			limit = { var:province_rank = 3 }
			set_city_status = $mapmode_type$_city_metropolis_$number$			
		}		
	}
}
